45  memset(&rgb, 0, sizeof rgb);
46 
47  if ((strcmp(argv[1], "help") == 0)) {

52  shell_print(shell, "'R..' indicates red is out of gamut.");

55

61

65  (unsigned int)ct, rgb.r, rgb.g, rgb.b);
66 
67  /* Display an out of gamut warning. */
