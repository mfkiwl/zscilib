117 int zsl_quat_mult(struct zsl_quat *qa, struct zsl_quat *qb,
118  struct zsl_quat *qm)
119 {
120  int rc = 0;
121 
122  qm->i = qa->r * qb->i + qa->i * qb->r + qa->j * qb->k - qa->k * qb->j;
123  qm->j = qa->r * qb->j - qa->i * qb->k + qa->j * qb->r + qa->k * qb->i;
124  qm->k = qa->r * qb->k + qa->i * qb->j - qa->j * qb->i + qa->k * qb->r;
125  qm->r = qa->r * qb->r - qa->i * qb->i - qa->j * qb->j - qa->k * qb->k;

130 int zsl_quat_exp(struct zsl_quat *q, struct zsl_quat *qe)
131 {
132  int rc = 0;
133 
134  ZSL_VECTOR_DEF(v, 3);
135  zsl_real_t vmag; /* Magnitude of v. */
136  zsl_real_t vsin; /* Sine of vm. */
137  zsl_real_t rexp; /* Exponent of q->r. */
138 
139  /* Populate the XYZ vector using ijk from q. */
140  v.data[0] = q->i;
141  v.data[1] = q->j;
142  v.data[2] = q->k;
143 
144  /* Calculate magnitude of v. */
145  vmag = zsl_vec_norm(&v);
146 
147  /* Normalise v to unit vector. */
148  zsl_vec_to_unit(&v);
149 
150  vsin = ZSL_SIN(vmag);
151  rexp = ZSL_EXP(q->r);
152 
153  qe->r = ZSL_COS(vmag) * rexp;
154  qe->i = v.data[0] * vsin * rexp;
155  qe->j = v.data[1] * vsin * rexp;
156  qe->k = v.data[2] * vsin * rexp;
157 
158  return rc;
159 }

274 int zsl_quat_slerp(struct zsl_quat *qa, struct zsl_quat *qb,
275  zsl_real_t t, struct zsl_quat *qi)
276 {
277  int rc = 0;
278  struct zsl_quat q1, q2; /* Interim quats. */
279  zsl_real_t dot; /* qa->r * qb->r. */
280  zsl_real_t phi; /* arccos(dot). */
281  zsl_real_t phi_s; /* sin(phi). */
282  zsl_real_t phi_st; /* sin(phi * (t)). */
283  zsl_real_t phi_smt; /* sin(phi * (1.0 - t)). */
284 
285  /*
286  * Unit quaternion slerp = qa * (qa^-1 * qb)^t
287  *
288  * We get there in a round-about way in this code, but we avoid pushing
289  * and popping values on the stack with trivial calls to helper functions.
290  */
291 
292  /* TODO: When t = 0.0 or t = 1.0, just memcpy qa or qb. */
293 
294  /* Make sure t is in a valid range. */
295  t = t < 0.0 ? 0.0 : t;
296  t = t > 1.0 ? 1.0 : t;
297 
298  dot = qa->r * qb->r;
299 
300  /*
301  * Slerp often has problems with angles close to zero. Consider handling
302  * those edge cases slightly differently?
303  */
304 
305  /* Calculate these once before-hand. */
306  phi = ZSL_ACOS(dot);
307  phi_s = ZSL_SIN(phi);
308  phi_st = ZSL_SIN(phi * t);
309  phi_smt = ZSL_SIN(phi * (1.0 - t));
310 
311  /* Calculate intermediate quats. */
312  q1.r = phi_smt / phi_s * qa->r;
313  q1.i = phi_smt / phi_s * qa->i;
314  q1.j = phi_smt / phi_s * qa->j;
315  q1.k = phi_smt / phi_s * qa->k;
316  q2.r = phi_st / phi_s * qb->r;
317  q2.i = phi_st / phi_s * qb->i;
318  q2.j = phi_st / phi_s * qb->j;
319  q2.k = phi_st / phi_s * qb->k;
320 
321  /* Final result = q1 + q2. */
322  qi->r = q1.r + q2.r;
323  qi->i = q1.i + q2.i;
324  qi->j = q1.j + q2.j;
325  qi->k = q1.k + q2.k;
326 
327  return rc;
328 }
